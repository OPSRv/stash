# stash-separator runtime deps. uv resolves these inside the venv at
# `Settings â†’ Separator â†’ Install` and the resulting interpreter is

# Notes on the pins:
# - `demucs==4.0.1`: the API surface main.py uses
#   (`demucs.pretrained` + `demucs.apply` + `demucs.audio`) is stable
#   from 4.0.1 onwards. The newer `demucs.api.Separator` wrapper is
#   only on git head and isn't in PyPI 4.0.1, so we don't depend on
#   it. Pinning explicitly takes resolver ambiguity off the table.
# - `librosa`: drives BPM detection via `librosa.beat.beat_track`
#   on the separated drums stem. We previously tried BeatNet, but
#   it imports `madmom` at module load and madmom's PyPI release is
#   stuck at 0.16.1 (2018) which doesn't build on Python â‰Ą3.10. The
#   only way to get a working madmom is git+URL, which is outside
#   the trust boundary of an opt-in installer running on user
#   machines. librosa's accuracy is ~80-90 % on cleaned drums, which
#   is good enough for the workflow.
# - `numpy<2`: librosa 0.10 still pins numpy<2; demucs has no upper
#   bound but is fine on 1.x.

demucs==4.0.1
librosa>=0.10,<0.12

# released against torch 2.x and `demucs.audio.save_audio` started
# requiring an external `torchcodec` package on torch â‰Ą2.7. Pinning
# 2.2.2 â€” the last version with the in-tree audio codec â€” sidesteps
# that without giving up Apple Silicon MPS support, which 2.2 has.
torch==2.2.2
torchaudio==2.2.2
numpy>=1.24,<2

# `coverage>=7` â€” workaround for a numba 0.65.1 import-time bug
# (`numba.misc.coverage_support` does `import coverage` and then
# `class NumbaTracer(coverage.types.Tracer):` at module load).
# Without an actual coverage package on sys.path, Python's PEP 420
# namespace lookup hands back an empty `coverage` module, the
# class statement raises `AttributeError: module 'coverage' has
# no attribute 'types'`, and every `import librosa` fails.
coverage>=7

# Spotify's basic-pitch â€” audio â†’ MIDI for the per-stem MIDI buttons in
# the Stems tab. Same model NeuralNote ships. Pure-Python wheel + a
# TF/ONNX dependency; the ONNX backend is the lightest of the three.
basic-pitch>=0.4
# basic-pitch (or one of its transitive deps â€” likely pretty_midi or
# librosa's audio backend) does a runtime `import pkg_resources`. That
# module ships with `setuptools`, which Python 3.11+ no longer
# auto-installs into fresh venvs. A bare `>=68` was tempting but
# `setuptools` 81.0.0 (2025-07) removed the `pkg_resources` module
# outright (it was deprecated in 80.x). Without an upper bound uv
# resolves to the newest matching wheel (~81+), basic-pitch's runtime
# `import pkg_resources` still crashes, and the user sees the same
# ModuleNotFoundError they saw before the pin was added. Cap at <81
# so we land on the last series that still ships the shim.
setuptools>=68,<81
